#. Write a C program to check whether given string is palindrome or not.#



#include <stdio.h>
#include<string.h>
int main()
{
    char str[20],str1[20];
    int i,j,len,flag;
    flag=0;
   printf(“enter the string:”);
    scanf("%s",str);
    len=strlen(str);
    j=0;
    for(i=len-1;i>=0;i--)
    {
       str1[j++]=str[i];
    }
    str1[j]='\0';
    for(i=0;i<len;i++)
    {
        if(str1[i]!=str[i])
        {
            flag=1;
            break;
        }
    }
    if(flag==0)
    {
        printf("palindrom");
    }
    else
    {
        printf("not palindrom");
    }
    return 0;
}
